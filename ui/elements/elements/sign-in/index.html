JS - Providers + flip + show password
// Optional: limit providers via data-auth-providers on 
// 
var base = document.documentElement.getAttribute('data-auth-base') || '/auth';
var allowed = document.documentElement.getAttribute('data-auth-providers');
var icons = document.documentElement.getAttribute('data-auth-icons');
var list = allowed ? allowed.split(',').map(function (p) { return p.trim().toLowerCase(); }) : null;

if (icons === 'false') {
  Array.from(document.querySelectorAll('.auth-provider-row--icons')).forEach(function (row) {
    row.style.display = 'none';
  });
}

Array.from(document.querySelectorAll('[data-auth-provider]')).forEach(function (btn) {
  var provider = btn.getAttribute('data-auth-provider');
  if (list && list.indexOf(provider) === -1) {
    btn.style.display = 'none';
    return;
  }
  btn.addEventListener('click', function () {
    window.location.href = base.replace(/\/$/, '') + '/' + provider;
  });
});

// Flip reset (Sign-in card)
var authCard = document.querySelector('[data-auth-card="signin"]');
if (authCard) {
  authCard.querySelector('[data-auth-forgot]').addEventListener('click', function () {
    authCard.classList.add('is-flipped');
  });
  authCard.querySelector('[data-auth-back]').addEventListener('click', function () {
    authCard.classList.remove('is-flipped');
  });
}

// Show / hide password
Array.from(document.querySelectorAll('[data-toggle-password]')).forEach(function (toggle) {
  var input = document.getElementById(toggle.getAttribute('data-toggle-password'));
  toggle.addEventListener('click', function () {
    var isPassword = input.type === 'password';
    input.type = isPassword ? 'text' : 'password';
    toggle.textContent = isPassword ? 'Hide' : 'Show';
  });
});

Quick notes
  • Set data-auth-base on <html> to change the provider URL base.
  • Limit providers with data-auth-providers="google,apple" on <html>.
  • The icon-only row is optional; hide it with data-auth-icons="false".
  • Replace the slide images in assets/ to customise the illustration.
  • Use data-auth-static="true" on the panel to disable carousel animation.
